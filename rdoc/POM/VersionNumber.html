     # File lib/pom/version_number.rb, line 205
205:     def self.parse_constraint( constraint )
206:       constraint = constraint.strip
207:       re = %{^(=~|~>|<=|>=|==|=|<|>)?\s*(\d+(:?[-.]\d+)*)$}
208:       if md = re.match( constraint )
209:         if op = md[1]
210:           op = '=~' if op == '~>'
211:           op = '==' if op == '='
212:           val = new( *md[2].split(/\W+/) )
213:         else
214:           op = '=='
215:           val = new( *constraint.split(/\W+/) )
216:         end
217:       else
218:         raise ArgumentError, "invalid constraint"
219:       end
220:       return op, val
221:     end
