    def read
      if @file
        text = File.read(file).strip
        text = text.sub(/\A[=]\s+(.*?)$/,'').strip
        scan = text.scan(/\=\=(.*?)\n(.*?)(^Changes:.*?)(?=\=\=|\Z)/m)
        scan.each do |header, notes, changes|
          @releases << Release.new(header, notes, changes)
        end
      end
    end
